# S3 with Terraform — Object Storage, Buckets, and Policies

## The Warehouse Analogy

Amazon S3 is like a globally distributed warehouse with unlimited capacity. Each bucket is a separate warehouse. Objects (files) are the items inside. You control who has the keys (bucket policy), whether items are versioned (versioning), how long to store them (lifecycle rules), and whether the warehouse doors face the public street (public access).

Terraform lets you provision and configure S3 buckets with all their features in code — so you never have to click through the console again.

## Basic S3 Bucket

```hcl
resource "aws_s3_bucket" "app_data" {
  bucket = "${var.project}-${var.environment}-data-${data.aws_caller_identity.current.account_id}"
  # Including account ID makes the name globally unique

  tags = {
    Name        = "${var.project}-${var.environment}-data"
    Environment = var.environment
    Project     = var.project
  }
}

data "aws_caller_identity" "current" {}
```

**Note:** S3 bucket names must be globally unique across ALL AWS accounts. Include your account ID or a random suffix.

## Versioning

```hcl
resource "aws_s3_bucket_versioning" "app_data" {
  bucket = aws_s3_bucket.app_data.id

  versioning_configuration {
    status = "Enabled"   # "Enabled" or "Suspended"
  }
}
```

With versioning enabled, deleting or overwriting an object just adds a new version. You can restore previous versions if something goes wrong.

## Encryption at Rest

```hcl
resource "aws_s3_bucket_server_side_encryption_configuration" "app_data" {
  bucket = aws_s3_bucket.app_data.id

  rule {
    apply_server_side_encryption_by_default {
      sse_algorithm = "AES256"   # or "aws:kms" for KMS encryption
    }
    bucket_key_enabled = true   # reduces KMS costs when using aws:kms
  }
}

# For KMS encryption (stronger, auditable)
resource "aws_s3_bucket_server_side_encryption_configuration" "app_kms" {
  bucket = aws_s3_bucket.app_data.id

  rule {
    apply_server_side_encryption_by_default {
      sse_algorithm     = "aws:kms"
      kms_master_key_id = aws_kms_key.s3.arn   # your KMS key
    }
    bucket_key_enabled = true
  }
}
```

## Block Public Access (Always Do This!)

```hcl
resource "aws_s3_bucket_public_access_block" "app_data" {
  bucket = aws_s3_bucket.app_data.id

  block_public_acls       = true
  block_public_policy     = true
  ignore_public_acls      = true
  restrict_public_buckets = true
}
```

Set all four to `true` for any bucket that should NOT be publicly accessible (most buckets). This overrides any bucket policy or ACL that might accidentally grant public access.

## Bucket Policy

```hcl
resource "aws_s3_bucket_policy" "app_data" {
  bucket = aws_s3_bucket.app_data.id

  # Must apply public access block BEFORE bucket policy if blocking public
  depends_on = [aws_s3_bucket_public_access_block.app_data]

  policy = jsonencode({
    Version = "2012-10-17"
    Statement = [
      {
        # Allow a specific IAM role to read objects
        Sid       = "AllowRoleReadAccess"
        Effect    = "Allow"
        Principal = {
          AWS = aws_iam_role.app.arn
        }
        Action   = ["s3:GetObject", "s3:ListBucket"]
        Resource = [
          aws_s3_bucket.app_data.arn,
          "${aws_s3_bucket.app_data.arn}/*"
        ]
      },
      {
        # Deny all non-HTTPS access
        Sid       = "DenyNonHTTPS"
        Effect    = "Deny"
        Principal = "*"
        Action    = "s3:*"
        Resource  = [
          aws_s3_bucket.app_data.arn,
          "${aws_s3_bucket.app_data.arn}/*"
        ]
        Condition = {
          Bool = {
            "aws:SecureTransport" = "false"
          }
        }
      }
    ]
  })
}
```

## Lifecycle Rules

Lifecycle rules automatically transition objects to cheaper storage classes or delete them after a set time — great for logs and backups:

```hcl
resource "aws_s3_bucket_lifecycle_configuration" "app_logs" {
  bucket = aws_s3_bucket.app_logs.id

  rule {
    id     = "log-retention"
    status = "Enabled"

    # Apply to all objects with "logs/" prefix
    filter {
      prefix = "logs/"
    }

    # Transition to cheaper storage after 30 days
    transition {
      days          = 30
      storage_class = "STANDARD_IA"   # Infrequent Access
    }

    # Transition to Glacier after 90 days
    transition {
      days          = 90
      storage_class = "GLACIER"
    }

    # Delete after 365 days
    expiration {
      days = 365
    }
  }

  rule {
    id     = "delete-old-versions"
    status = "Enabled"

    filter {}   # apply to all objects

    # Delete non-current versions after 30 days
    noncurrent_version_expiration {
      noncurrent_days = 30
    }

    # Delete incomplete multipart uploads after 7 days
    abort_incomplete_multipart_upload {
      days_after_initiation = 7
    }
  }
}
```

## Static Website Hosting

```hcl
resource "aws_s3_bucket" "website" {
  bucket = "${var.project}-website"
}

# For static websites, you need to allow public access
resource "aws_s3_bucket_public_access_block" "website" {
  bucket                  = aws_s3_bucket.website.id
  block_public_acls       = false
  block_public_policy     = false
  ignore_public_acls      = false
  restrict_public_buckets = false
}

resource "aws_s3_bucket_website_configuration" "website" {
  bucket = aws_s3_bucket.website.id

  index_document {
    suffix = "index.html"
  }

  error_document {
    key = "error.html"
  }
}

# Bucket policy to allow public read
resource "aws_s3_bucket_policy" "website" {
  bucket     = aws_s3_bucket.website.id
  depends_on = [aws_s3_bucket_public_access_block.website]

  policy = jsonencode({
    Version = "2012-10-17"
    Statement = [{
      Sid       = "PublicReadGetObject"
      Effect    = "Allow"
      Principal = "*"
      Action    = "s3:GetObject"
      Resource  = "${aws_s3_bucket.website.arn}/*"
    }]
  })
}

output "website_url" {
  value = aws_s3_bucket_website_configuration.website.website_endpoint
}
```

## CORS Configuration (for Web Apps)

```hcl
resource "aws_s3_bucket_cors_configuration" "assets" {
  bucket = aws_s3_bucket.assets.id

  cors_rule {
    allowed_headers = ["*"]
    allowed_methods = ["GET", "HEAD"]
    allowed_origins = ["https://myapp.com", "https://www.myapp.com"]
    expose_headers  = ["ETag"]
    max_age_seconds = 3000
  }
}
```

## Complete S3 Setup for Application Data

```hcl
locals {
  bucket_name = "${var.project}-${var.environment}-data-${data.aws_caller_identity.current.account_id}"
}

resource "aws_s3_bucket" "app_data"                                 { bucket = local.bucket_name }
resource "aws_s3_bucket_versioning" "app_data"                      { bucket = aws_s3_bucket.app_data.id; versioning_configuration { status = "Enabled" } }
resource "aws_s3_bucket_server_side_encryption_configuration" "app_data" {
  bucket = aws_s3_bucket.app_data.id
  rule { apply_server_side_encryption_by_default { sse_algorithm = "AES256" } }
}
resource "aws_s3_bucket_public_access_block" "app_data" {
  bucket                  = aws_s3_bucket.app_data.id
  block_public_acls       = true
  block_public_policy     = true
  ignore_public_acls      = true
  restrict_public_buckets = true
}

output "bucket_name" { value = aws_s3_bucket.app_data.id }
output "bucket_arn"  { value = aws_s3_bucket.app_data.arn }
```
